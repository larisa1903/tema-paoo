public class PrimitiveExample {
    public static void main(String[] args) {
        int x = 10;  // Declarație simplă
        int y = x + 5;  // Operații rapide
        System.out.println(y);  // 15
        // int e mai rapid si nu poate fi null
        // la operatii matematice

        Integer a = 10;
        Integer b = null; // Permis, spre deosebire de int
        // integer e mai lent si poate fi null
        //array si map
        System.out.println(a + 5); // afișează 15
    }
}
